CMD Calculator

@echo off
title Calc
:loop
cls
set /p first=First number:
echo.
echo Mark (+ - x /)
set /p mark=Mark:
echo.
set /p second=Second number:
set /a final=%first%%mark%%second%
echo.
echo %final%
pause
goto loop
